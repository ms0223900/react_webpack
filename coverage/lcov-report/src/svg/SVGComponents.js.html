const HalfCircle = (props) => {
  const { pos1: [x1, y1], dir=0, r=20, clockDir=0 } = props
  const dxdy = [`0 -${r * 2}`, `${r * 2} 0`, `0 ${r * 2}`, `-${r * 2} 0`][dir]
  return (
    <path d={`
      M ${x1} ${y1} 
      a ${r} ${r}, 0, 0 ${clockDir}, ${dxdy}
      `}
      {...props} 
    />
  )
}

export const RoundedCorner = (props) => {
  const { x, y, h, v } = props
  let { r } = props
  // max radius
  if(r > Math.abs(h / 2)  || r > Math.abs(v / 2)) {
    if(h > v) {
      r = Math.abs(h / 2)
    }
    r = Math.abs(v / 2)
  }
  // to adjust the line direction, change r to vector r
  let 
    rh = h >= 0 ? r : -r, 
    rv = v >= 0 ? r : -r;
 
  return (
    <path 
      d={`
        M${x} ${y} 
        h${ h >= 0 ? h - r : h + r } 
        q${rh} 0, ${rh} ${rv} 
        v${ v >= 0 ? v - r * 2 : v + r * 2 } 
        q0 ${rv}, ${-rh} ${rv} 
        h${h >= 0 ? -(h - r) : -(h + r) } 
      `}
      fill={'none'} 
      stroke={'#000'} 
      strokeWidth={2.5}
      {...props} />
  )
}
